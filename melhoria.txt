1. A Lógica de Monitoramento (Backend/Service)
Este script faz o "diff" para identificar se uma conta saiu da contestação ou do BOV.
// types/ads.ts
export type AccountStatus = 'ENABLED' | 'SUSPENDED' | 'PENDING_VERIFICATION' | 'CLOSED';


interface AccountMetrics {
  customerId: string;
  previousStatus: AccountStatus;
  currentStatus: AccountStatus;
  hasOpenAppeal: boolean; // Marcado via banco de dados interno
  hasPendingBOV: boolean;  // Marcado via banco de dados interno
}


// services/recoveryTracker.ts
export const calculateRecoveryStats = (accounts: AccountMetrics[]) => {
  return accounts.reduce((stats, acc) => {
    
    // 1. Quantas voltaram da Contestação (Estava Suspensa e agora está Ativa)
    if (acc.previousStatus === 'SUSPENDED' && acc.currentStatus === 'ENABLED' && acc.hasOpenAppeal) {
      stats.recoveredFromAppeal++;
    }


    // 2. Quantas voltaram do BOV (Estava em Verificação e agora está Ativa)
    if (acc.previousStatus === 'PENDING_VERIFICATION' && acc.currentStatus === 'ENABLED' && acc.hasPendingBOV) {
      stats.recoveredFromBOV++;
    }


    // 3. Taxa de Sucesso (Cálculo de Eficiência)
    stats.appealSuccessRate = stats.totalAppeals > 0 
      ? (stats.recoveredFromAppeal / stats.totalAppeals) * 100 
      : 0;


    return stats;
  }, { 
    recoveredFromAppeal: 0, 
    recoveredFromBOV: 0, 
    totalAppeals: accounts.filter(a => a.hasOpenAppeal).length,
    appealSuccessRate: 0 
  });
};

   * 🔴 High Risk (Score < 50): Cliente que "mata" contas por erro de anúncio ou problemas de pagamento.
   * Ação do Sistema: Bloquear automaticamente a venda de contas G2 Premium para esse nível até que o score suba.

🛠️ Fluxo de Trabalho Otimizado (Standard Operating Procedure)
1. Etapa de Produção (Colaboradores)
   * Ação no Sistema: O colaborador marca as contas produzidas, gerando um Lote de Produção com IDs específicos.
   * Comunicação: Além do registro no ERP, o colaborador envia o link/ID do lote no grupo AdsPower Conferência Produção para alertar a gerência.
2. Etapa de Conferência e Estoque (Gerência de Produção)
   * Validação: O Gerente acessa o sistema, revisa os IDs e valida a qualidade.
   * Antes de exportar e importar para o sistema da Ads Ativos o gerente de produção deverá trocar as senhas dos e-mails para evitar roubo, essas novas senhas deverão ser anexadas no sistema para fácil copia e cola. 
   * Movimentação: * Exporta o arquivo de dados do AdsPower (configurações/cookies).
   * Importa no Estoque Digital do sistema.
   * Status do Lote: Alterado para "Disponível para Entrega".






3. Etapa de Venda e Entrega (Setor de Entrega)
   * Separação: Ao receber um pedido, o setor baixa as contas do estoque conforme a especificação do cliente.
   * Preparação: Importa no AdsPower do setor de entrega para uma conferência final rápida.
   * Finalização: Envio do lote ao cliente e atualização do status para "Entregue".
4. Gestão de Qualidade (Logística Reversa)
   * Reporte de Erros: Caso o setor de entrega identifique contas inválidas ou com problemas, deve registrar no sistema o ID da Conta + Tipo de Erro.
   * Escritório Central: O Administrador Geral recebe um alerta automático para auditoria (substituição de conta ou correção do colaborador).

<script>
window.dataLayer = window.dataLayer || [];
function trackWhatsAppJivo() {
  document.addEventListener('click', function(e) {
    var el = e.target;
    while (el && el !== document.body) {
      if (el.tagName === 'A' && el.href && (el.href.includes('wa.me') || el.href.includes('api.whatsapp.com'))) {
        window.dataLayer.push({
          event: 'whatsapp_click',
          contact_method: 'whatsapp',
          source: 'jivochat'
        });
        break;
      }
      el = el.parentElement;
    }
  });
}
if (document.readyState === 'loading') {
  document.addEventListener('DOMContentLoaded', trackWhatsAppJivo);
} else {
  trackWhatsAppJivo();
}
</script>
